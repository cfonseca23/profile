## Arquitectura de Datos

### Modelos C#

```csharp
// Models/CvData.cs
public class CvData
{
    public PersonalInfo Personal { get; set; }
    public string Summary { get; set; }  // Template con {years}
    public List<Skill> Skills { get; set; }
    public List<Experience> Experiences { get; set; }
    public List<Education> Education { get; set; }
    public List<string> Certifications { get; set; }
    
    // Calculado
    public int YearsOfExperience => 
        DateTime.Now.Year - Experiences.Min(e => e.StartYear);
    
    public string GetSummary() => 
        Summary.Replace("{years}", YearsOfExperience.ToString());
}

// Models/Experience.cs
public class Experience
{
    public int StartYear { get; set; }
    public int? EndYear { get; set; }  // null = Presente
    public string Company { get; set; }
    public string Role { get; set; }
    public string Industry { get; set; }
    public List<string> Technologies { get; set; }
    
    public string Period => EndYear.HasValue 
        ? $"{StartYear} - {EndYear}" 
        : $"{StartYear} - Presente";
}
```
